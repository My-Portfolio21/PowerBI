SELECT
    w.query,
    w.queue_start_time,
    w.service_class,
    w.total_exec_time,
    s.service_class,
    s.total_exec_time AS total_exec_time_stuck,
    s.wait_event
FROM stl_wlm_query w
JOIN stl_wlm_query_state s
ON w.query = s.query
WHERE w.status = 'Aborted'
AND s.queue_start_time < (GETDATE() - INTERVAL '1 hour')
AND w.pid = <Your_PID>;
