SELECT
  l.locktype,
  l.database,
  l.relation::regclass,
  l.transaction AS blocking_transaction,
  l.mode,
  a.usename AS blocked_user,
  a.current_query AS blocked_query
FROM pg_locks l
JOIN pg_stat_activity a ON l.pid = a.procpid
WHERE a.stl_starttime >= '2023-10-18 12:00:00' AND a.stl_starttime <= '2023-10-18 17:00:00'
AND a.last_exec_status = 'Abort';
